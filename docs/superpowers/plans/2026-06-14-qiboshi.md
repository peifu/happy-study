        // ========== 七博士 Chat Application ==========

        (function() {
            // --- State ---
            var messages = [];
            var isLoading = false;

            // --- DOM refs ---
            var setupPrompt = document.getElementById('setupPrompt');
            var chatUI = document.getElementById('chatUI');
            var chatMessages = document.getElementById('chatMessages');
            var welcomeBanner = document.getElementById('welcomeBanner');
            var chatInput = document.getElementById('chatInput');
            var sendBtn = document.getElementById('sendBtn');
            var voiceBtn = document.getElementById('voiceBtn');
            var settingsOverlay = document.getElementById('settingsOverlay');
            var toast = document.getElementById('toast');
            var currentUtterance = null;
            var recognition = null;
            var isListening = false;

            // --- Helpers ---
            function g(key, fallback) {
                return localStorage.getItem(key) || fallback;
            }

            function s(key, val) {
                localStorage.setItem(key, val);
            }

            function getApiKey() {
                return g('deepseek_api_key', '');
            }

            function getModel() {
                return g('qiboshi_model', 'deepseek-chat');
            }

            function loadHistory() {
                try { return JSON.parse(g('qiboshi_history', '[]')); } catch(e) { return []; }
            }

            function saveHistory(msgs) {
                var recent = msgs.slice(-50);
                s('qiboshi_history', JSON.stringify(recent));
            }

            function scrollBottom() {
                chatMessages.scrollTop = chatMessages.scrollHeight;
            }

            function showToast(text) {
                toast.textContent = text;
                toast.classList.add('show');
                clearTimeout(toast._t);
                toast._t = setTimeout(function() { toast.classList.remove('show'); }, 2500);
            }

            // --- Markdown rendering ---
            function renderMarkdown(text) {
                var escaped = text
                    .replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;');
                // code blocks ```
                escaped = escaped.replace(/```(\w*)\n([\s\S]*?)```/g,
                    '<pre><code>$2</code></pre>');
                // inline code `
                escaped = escaped.replace(/`([^`]+)`/g, '<code>$1</code>');
                // bold **
                escaped = escaped.replace(/\*\*([^*]+)\*\*/g, '<strong>$1</strong>');
                // unordered lists
                escaped = escaped.replace(/^- (.+)$/gm, '<li>$1</li>');
                escaped = escaped.replace(/(<li>.*<\/li>)/s, '<ul>$1</ul>');
                // numbered lists
                escaped = escaped.replace(/^\d+\. (.+)$/gm, '<li>$1</li>');
                // newlines to <br>
                escaped = escaped.replace(/\n/g, '<br>');
                return escaped;
            }

            // --- Messages ---
            function appendUserMsg(text) {
                var div = document.createElement('div');
                div.className = 'msg-bubble msg-user';
                div.textContent = text;
                chatMessages.appendChild(div);
                messages.push({ role: 'user', content: text });
                saveHistory(messages);
                hideWelcome();
                scrollBottom();
            }

            function appendAIMsg(text) {
                var div = document.createElement('div');
                div.className = 'msg-bubble msg-ai';
                div.innerHTML = renderMarkdown(text);
                var speakBtn = document.createElement('button');
                speakBtn.className = 'speak-msg-btn';
                speakBtn.innerHTML = '<i class="fas fa-volume-up"></i> 朗读';
                speakBtn.addEventListener('click', function() {
                    speakText(text, speakBtn);
                });
                div.appendChild(speakBtn);
                chatMessages.appendChild(div);
                messages.push({ role: 'assistant', content: text });
                saveHistory(messages);
                scrollBottom();
            }

            function appendError(text, retryFn) {
                var div = document.createElement('div');
                div.className = 'msg-error';
                div.textContent = text;
                if (retryFn) {
                    var btn = document.createElement('button');
                    btn.className = 'retry-btn';
                    btn.textContent = '重试';
                    btn.addEventListener('click', retryFn);
                    div.appendChild(btn);
                }
                chatMessages.appendChild(div);
                scrollBottom();
            }

            function showTyping() {
                var div = document.createElement('div');
                div.className = 'typing-indicator';
                div.id = 'typingIndicator';

            }

            function hideTyping() {

                if (el) el.remove();
            }

            function hideWelcome() {
                if (welcomeBanner) welcomeBanner.style.display = 'none';
            }

            // --- DeepSeek API ---
            function callDeepSeek(userMsg, retryCount) {
                retryCount = retryCount || 0;
                isLoading = true;
                sendBtn.disabled = true;
                showTyping();

                var apiKey = getApiKey();
                var model = getModel();

                var apiMessages = [{
                    role: 'system',
                    content: '你是七博士，一位专门帮助小学生学习的AI助手。你善于用通俗易懂的方式解释知识，鼓励学生思考。请用中文回答，保持友好、鼓励的语气。回答要简洁明了。'
                }];
                var recent = messages.slice(-20);
                apiMessages = apiMessages.concat(recent);

                fetch('https://api.deepseek.com/v1/chat/completions', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'Authorization': 'Bearer ' + apiKey
                    },
                    body: JSON.stringify({
                        model: model,
                        messages: apiMessages,
                        max_tokens: 1024,
                        temperature: 0.7
                    })
                })
                .then(function(res) {
                    if (!res.ok) {
                        return res.json().then(function(err) {
                            throw { status: res.status, body: err };
                        });
                    }
                    return res.json();
                })
                .then(function(data) {
                    hideTyping();
                    var reply = data.choices[0].message.content;
                    appendAIMsg(reply);
                    isLoading = false;
                    sendBtn.disabled = false;
                    chatInput.focus();
                })
                .catch(function(err) {
                    hideTyping();
                    isLoading = false;
                    sendBtn.disabled = false;
                    var status = err.status || 0;

                    if (status === 401) {
                        appendError('API Key 无效，请在设置中更新。', function() { openSettings(); });
                    } else if (status === 429) {
                        appendError('AI博士有点累了，请稍后再试~');
                    } else if (status === 402) {
                        appendError('API 余额不足，请检查你的 DeepSeek 账户。');
                    } else {
                        var msg = '连接失败：' + (err.message || '未知错误');
                        appendError(msg, function() { retrySend(userMsg); });
                    }
                });
            }

            function retrySend(userMsg) {
                messages.pop();
                saveHistory(messages);
                callDeepSeek(userMsg);
            }

            function sendMessage() {
                var text = chatInput.value.trim();
                if (!text || isLoading) return;
                chatInput.value = '';
                appendUserMsg(text);
                callDeepSeek(text);
            }

            // --- Voice Input (SpeechRecognition) ---
            function initSpeechRecognition() {
                var SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
                if (!SpeechRecognition) {
                    voiceBtn.style.display = 'none';
                    return;
                }
                recognition = new SpeechRecognition();
                recognition.lang = 'zh-CN';
                recognition.interimResults = false;
                recognition.continuous = false;

                recognition.addEventListener('result', function(e) {
                    var transcript = e.results[0][0].transcript;
                    chatInput.value = transcript;
                    chatInput.focus();
                    stopListening();
                });

                recognition.addEventListener('error', function(e) {
                    stopListening();
                    if (e.error === 'not-allowed') {
                        showToast('麦克风权限被拒绝');
                    } else if (e.error !== 'aborted') {
                        showToast('语音识别出错：' + e.error);
                    }
                });

                recognition.addEventListener('end', function() {
                    stopListening();
                });
            }

            function startListening() {
                if (!recognition || isListening) return;
                isListening = true;
                voiceBtn.classList.add('listening');
                voiceBtn.querySelector('i').className = 'fas fa-microphone-alt';
                try { recognition.start(); } catch(e) {}
            }

            function stopListening() {
                isListening = false;
                voiceBtn.classList.remove('listening');
                voiceBtn.querySelector('i').className = 'fas fa-microphone';
                try { recognition.stop(); } catch(e) {}
            }

            // --- Voice Output (SpeechSynthesis) ---
            function speakText(text, btn) {
                if (currentUtterance) {
                    speechSynthesis.cancel();
                    if (currentUtterance._btn) {
                        currentUtterance._btn.classList.remove('playing');
                        currentUtterance._btn.querySelector('i').className = 'fas fa-volume-up';
                        currentUtterance._btn.innerHTML = '<i class="fas fa-volume-up"></i> 朗读';
                    }
                    currentUtterance = null;
                    if (btn && btn.classList.contains('playing')) return;
                }

                var utterance = new SpeechSynthesisUtterance(text);
                utterance.lang = 'zh-CN';
                utterance.rate = parseFloat(g('speechRateZh', '1.0'));
                utterance.pitch = parseFloat(g('speechPitchZh', '1.0'));

                var savedVoiceName = g('selectedVoiceZh', '');
                var voices = speechSynthesis.getVoices();
                if (savedVoiceName) {
                    var v = voices.find(function(voice) { return voice.name === savedVoiceName; });
                    if (v) utterance.voice = v;
                }
                if (!utterance.voice) {
                    var cv = voices.find(function(voice) { return voice.lang.includes('zh'); });
                    if (cv) utterance.voice = cv;
                }

                utterance._btn = btn;
                currentUtterance = utterance;

                if (btn) {
                    btn.classList.add('playing');
                    btn.querySelector('i').className = 'fas fa-volume-up';
                    btn.innerHTML = '<i class="fas fa-spinner fa-pulse"></i> 播放中';
                }

                utterance.addEventListener('end', function() {
                    if (btn) {
                        btn.classList.remove('playing');
                        btn.querySelector('i').className = 'fas fa-volume-up';
                        btn.innerHTML = '<i class="fas fa-volume-up"></i> 朗读';
                    }
                    currentUtterance = null;
                });

                utterance.addEventListener('error', function() {
                    if (btn) {
                        btn.classList.remove('playing');
                        btn.querySelector('i').className = 'fas fa-volume-up';
                        btn.innerHTML = '<i class="fas fa-volume-up"></i> 朗读';
                    }
                    currentUtterance = null;
                });

                speechSynthesis.speak(utterance);
            }

            // Preload voices
            speechSynthesis.getVoices();
            speechSynthesis.addEventListener('voiceschanged', function() { speechSynthesis.getVoices(); });

            // --- Settings ---
            function openSettings() {
                document.getElementById('settingsApiKey').value = getApiKey();
                document.getElementById('settingsModel').value = getModel();
                settingsOverlay.classList.add('open');
            }

            function closeSettings() {
                settingsOverlay.classList.remove('open');
            }

            function saveSettings() {
                var newKey = document.getElementById('settingsApiKey').value.trim();
                var newModel = document.getElementById('settingsModel').value.trim() || 'deepseek-chat';
                s('deepseek_api_key', newKey);
                s('qiboshi_model', newModel);
                closeSettings();
                showToast('设置已保存');

                if (newKey && chatUI.style.display === 'none') {
                    setupPrompt.style.display = 'none';
                    chatUI.style.display = 'flex';
                    restoreChat();
                }
            }

            function clearHistory() {
                messages = [];
                s('qiboshi_history', '[]');
                chatMessages.innerHTML = '';
                // Restore welcome banner
                var banner = document.createElement('div');
                banner.className = 'welcome-banner';
                banner.id = 'welcomeBanner';
                banner.innerHTML = '<div class="welcome-icon">🤖</div>' +
                    '<h2>你好！我是七博士</h2>' +
                    '<p>有什么学习问题想问我？</p>' +
                    '<div class="quick-questions">' +
                        '<span class="quick-q" data-q="光合作用是什么？">光合作用是什么？</span>' +
                        '<span class="quick-q" data-q="如何快速记住英语单词？">如何快速记住英语单词？</span>' +
                        '<span class="quick-q" data-q="鸡兔同笼问题怎么解？">鸡兔同笼问题怎么解？</span>' +
                    '</div>';
                chatMessages.appendChild(banner);
                welcomeBanner = document.getElementById('welcomeBanner');
                bindQuickQuestions();
                closeSettings();
                showToast('对话已清除');
            }

            // --- Restore chat from history ---
            function restoreChat() {
                messages = loadHistory();
                if (messages.length > 0) {
                    hideWelcome();
                    messages.forEach(function(msg) {
                        var div = document.createElement('div');
                        if (msg.role === 'user') {
                            div.className = 'msg-bubble msg-user';
                            div.textContent = msg.content;
                        } else {
                            div.className = 'msg-bubble msg-ai';
                            div.innerHTML = renderMarkdown(msg.content);
                            var speakBtn = document.createElement('button');
                            speakBtn.className = 'speak-msg-btn';
                            speakBtn.innerHTML = '<i class="fas fa-volume-up"></i> 朗读';
                            speakBtn.addEventListener('click', function() {
                                speakText(msg.content, speakBtn);
                            });
                            div.appendChild(speakBtn);
                        }
                        chatMessages.appendChild(div);
                    });
                    scrollBottom();
                }
            }

            // --- Quick questions ---
            function bindQuickQuestions() {
                var qqs = document.querySelectorAll('.quick-q');
                qqs.forEach(function(el) {
                    el.addEventListener('click', function() {
                        var q = this.getAttribute('data-q');
                        chatInput.value = q;
                        sendMessage();
                    });
                });
            }

            // --- Event bindings ---
            sendBtn.addEventListener('click', sendMessage);
            chatInput.addEventListener('keydown', function(e) {
                if (e.key === 'Enter' && !e.shiftKey) {
                    e.preventDefault();
                    sendMessage();
                }
            });

            voiceBtn.addEventListener('click', function() {
                if (isListening) {
                    stopListening();
                } else {
                    startListening();
                }
            });

            document.getElementById('backBtn').addEventListener('click', function() {
                window.location.href = 'index.html';
            });

            document.getElementById('settingsBtn').addEventListener('click', openSettings);
            document.getElementById('closeSettingsBtn').addEventListener('click', closeSettings);
            document.getElementById('saveSettingsBtn').addEventListener('click', saveSettings);
            document.getElementById('clearHistoryBtn').addEventListener('click', function() {
                if (confirm('确定要清除所有对话记录吗？')) {
                    clearHistory();
                }
            });

            document.getElementById('saveKeyBtn').addEventListener('click', function() {
                var key = document.getElementById('apiKeyInput').value.trim();
                if (!key) { showToast('请输入 API Key'); return; }
                s('deepseek_api_key', key);
                setupPrompt.style.display = 'none';
                chatUI.style.display = 'flex';
                restoreChat();
                showToast('欢迎使用七博士！');
            });

            settingsOverlay.addEventListener('click', function(e) {
                if (e.target === settingsOverlay) closeSettings();
            });

            bindQuickQuestions();

            // --- Init ---
            syncThemeFromIndex();
            initSpeechRecognition();

            if (getApiKey()) {
                setupPrompt.style.display = 'none';
                chatUI.style.display = 'flex';
                restoreChat();
            }

            // Nav & assistant injection
            var NAV_CONFIG = [
                { page: 'chat', icon: 'fa-robot', label: 'AI对话', active: true }
            ];
            injectNav(NAV_CONFIG);
            injectFooter();
            injectAssistant();
            setupHomeLink('homeLink', 'index.html');
            initLearningAssistant();
        })();
    </script>
</body>
